  void TMyApp::handleEvent(TEvent& event)
  {

     TApplication::handleEvent( event );

     if( event.what == evCommand )
     {
        switch( event.message.command)
        {
        case cmAbout:
          messageBox(about_message, mfInformation | mfOKButton );
          clearEvent( event );
          break;

        default:
          break;
        }
     }
  }
